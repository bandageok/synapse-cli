```typescript
// src/commands/registry.ts
export interface SlashCommand {
  name: string;
  description: string;
  usage?: string;
  handler: (args: string, deps: REPLDeps) => Promise<void>;
}

export class CommandRegistry {
  private commands = new Map<string, SlashCommand>();

  register(cmd: SlashCommand): void {
    this.commands.set(cmd.name, cmd);
  }

  get(name: string): SlashCommand | undefined {
    return this.commands.get(name);
  }

  list(): SlashCommand[] {
    return Array.from(this.commands.values());
  }

  async execute(input: string, deps: REPLDeps): Promise<boolean> {
    if (!input.startsWith('/')) return false;
    const [name, ...args] = input.slice(1).split(' ');
    const cmd = this.commands.get(name);
    if (!cmd) return false;
    await cmd.handler(args.join(' '), deps);
    return true;
  }
}
```

| `/model [name]` | 查看/切换模型 | 读/写 provider.model |
| `/memory` | 查看 MEMORY.md | 读文件并显示 |
| `/soul` | 查看 SOUL.md | 读文件并显示 |
